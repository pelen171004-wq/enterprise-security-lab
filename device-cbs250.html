CBS CLI Configuration

CBS CLI
SW-A — VLAN Database + Trunk + Access ports 
SW-A> enable
SW-A# configure terminal

! VLAN Database
SW-A(config)# vlan 20
SW-A(config-vlan)#  name USERS
SW-A(config-vlan)# exit
SW-A(config)# vlan 30
SW-A(config-vlan)#  name MGMT
SW-A(config-vlan)# exit
SW-A(config)# vlan 60
SW-A(config-vlan)#  name GUEST
SW-A(config-vlan)# exit

! Trunk gi1 → FortiGate 60E
SW-A(config)# interface gi1
SW-A(config-if)#  switchport mode trunk
SW-A(config-if)#  switchport trunk allowed vlan add 20,30,60
SW-A(config-if)# exit

! Trunk gi4 → SW-B
SW-A(config)# interface gi4
SW-A(config-if)#  switchport mode trunk
SW-A(config-if)#  switchport trunk allowed vlan add 20,30,60
SW-A(config-if)# exit

! Access gi2–gi3 → VLAN30 Management
SW-A(config)# interface gi2
SW-A(config-if)#  switchport mode access
SW-A(config-if)#  switchport access vlan 30
SW-A(config-if)# exit
SW-A(config)# interface gi3
SW-A(config-if)#  switchport mode access
SW-A(config-if)#  switchport access vlan 30
SW-A(config-if)# exit

! Access gi5 → VLAN20 Users
SW-A(config)# interface gi5
SW-A(config-if)#  switchport mode access
SW-A(config-if)#  switchport access vlan 20
SW-A(config-if)# exit

! Access gi6 → VLAN60 Guest
SW-A(config)# interface gi6
SW-A(config-if)#  switchport mode access
SW-A(config-if)#  switchport access vlan 60
SW-A(config-if)# exit

! Management IP na VLAN30
SW-A(config)# interface vlan 30
SW-A(config-if)#  ip address 10.10.30.2 255.255.255.0
SW-A(config-if)# exit
SW-A(config)# ip default-gateway 10.10.30.1

SW-A(config)# end
SW-A# copy running-config startup-config
